160  function xi2 = quinlantest(o, Xj, classes, threshold)
161  m = numel(Xj);
162  ELAB = classes2boolean(classes);
163  L = sum(ELAB(Xj <= threshold,:),1) + 0.001;
164  R = sum(ELAB(Xj > threshold,:),1) + 0.001;
165  LL = (L+R) * sum(L) / m;
166  RR = (L+R) * sum(R) / m;
167  xi2 = sum(((L-LL).^2)./LL + ((R-RR).^2)./RR);
168  end;
