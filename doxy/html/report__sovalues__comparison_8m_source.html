202  function v = some_items(o, nar, choiceidx)
203  if isempty(choiceidx)
204  v = 1:o.maxrows;
205  else
206  A = .4;

212  i1 = floor(choiceidx-share2/2*nit);

221

224  i1 = i1-dif;
225  i2 = i2-dif;
226  end;
